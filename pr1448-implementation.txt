+using Aspire.Hosting.ApplicationModel;
+using Aspire.Hosting.Lifecycle;
+using Microsoft.Extensions.Logging;
+using System.Diagnostics;
+
+namespace Aspire.Hosting;
+
+/// <summary>
+/// Manages the Kind cluster lifecycle: creates clusters before the application starts
+/// and deletes them when it stops.
+/// </summary>
+internal sealed class KindClusterLifecycleHook : IDistributedApplicationLifecycleHook
+{
+    private readonly ILogger<KindClusterLifecycleHook> _logger;
+
+    public KindClusterLifecycleHook(ILogger<KindClusterLifecycleHook> logger)
+    {
+        _logger = logger;
+    }
+
+    /// <summary>
+    /// Creates all Kind clusters before the distributed application starts.
+    /// Runs: <c>kind create cluster --name {clusterName} --kubeconfig {kubeconfigPath}</c>
+    /// </summary>
+    public async Task BeforeStartAsync(DistributedApplicationModel appModel, CancellationToken cancellationToken = default)
+    {
+        var clusters = appModel.Resources.OfType<KindClusterResource>().ToList();
+        if (clusters.Count == 0)
+        {
+            return;
+        }
+
+        foreach (var cluster in clusters)
+        {
+            _logger.LogInformation("Creating Kind cluster '{ClusterName}' (kubeconfig: {KubeconfigPath})",
+                cluster.ClusterName, cluster.KubeconfigPath);
+
+            var exitCode = await RunCommandAsync(
+                "kind",
+                $"create cluster --name {cluster.ClusterName} --kubeconfig \"{cluster.KubeconfigPath}\"",
+                cancellationToken);
+
+            if (exitCode != 0)
+            {
+                throw new InvalidOperationException(
+                    $"Failed to create Kind cluster '{cluster.ClusterName}'. " +
+                    $"Ensure 'kind' is installed and Docker is running.");
+            }
+
+            _logger.LogInformation("Kind cluster '{ClusterName}' created successfully.", cluster.ClusterName);
+        }
+    }
+
+    /// <summary>
+    /// Runs a basic health check (<c>kubectl get nodes</c>) after resources are created.
+    /// </summary>
+    public async Task AfterResourcesCreatedAsync(DistributedApplicationModel appModel, CancellationToken cancellationToken = default)
+    {
+        var clusters = appModel.Resources.OfType<KindClusterResource>().ToList();
+        if (clusters.Count == 0)
+        {
+            return;
+        }
+
+        foreach (var cluster in clusters)
+        {
+            _logger.LogInformation("Running health check for Kind cluster '{ClusterName}'.", cluster.ClusterName);
+
+            var exitCode = await RunCommandAsync(
+                "kubectl",
+                $"get nodes --kubeconfig \"{cluster.KubeconfigPath}\"",
+                cancellationToken);
+
+            if (exitCode != 0)
+            {
+                _logger.LogWarning(
+                    "Health check failed for Kind cluster '{ClusterName}'. " +
+                    "The cluster may still be initializing.",
+                    cluster.ClusterName);
+            }
+            else
+            {
+                _logger.LogInformation("Kind cluster '{ClusterName}' is healthy.", cluster.ClusterName);
+            }
+        }
+    }
+
+    /// <summary>
+    /// Deletes all Kind clusters before the distributed application stops.
+    /// Runs: <c>kind delete cluster --name {clusterName}</c>
+    /// </summary>
+    public async Task BeforeStopAsync(DistributedApplicationModel appModel, CancellationToken cancellationToken = default)
+    {
+        var clusters = appModel.Resources.OfType<KindClusterResource>().ToList();
+        if (clusters.Count == 0)
+        {
+            return;
+        }
+
+        foreach (var cluster in clusters)
+        {
+            _logger.LogInformation("Deleting Kind cluster '{ClusterName}'.", cluster.ClusterName);
+
+            var exitCode = await RunCommandAsync(
+                "kind",
+                $"delete cluster --name {cluster.ClusterName}",
+                cancellationToken);
+
+            if (exitCode != 0)
+            {
+                // Log a warning rather than throwing — stop should be best-effort.
+                _logger.LogWarning(
+                    "Failed to delete Kind cluster '{ClusterName}' (exit code {ExitCode}). " +
+                    "You may need to run 'kind delete cluster --name {ClusterName}' manually.",
+                    cluster.ClusterName, exitCode, cluster.ClusterName);
+            }
+            else
+            {
+                _logger.LogInformation("Kind cluster '{ClusterName}' deleted.", cluster.ClusterName);
+            }
+        }
+    }
+
+    private static async Task<int> RunCommandAsync(string fileName, string arguments, CancellationToken cancellationToken)
+    {
+        var psi = new ProcessStartInfo
+        {
+            FileName = fileName,
+            Arguments = arguments,
+            RedirectStandardOutput = true,
+            RedirectStandardError = true,
+            UseShellExecute = false,
+            CreateNoWindow = true,
+        };
+
+        using var process = new Process { StartInfo = psi };
+        process.Start();
+
+        await process.WaitForExitAsync(cancellationToken);
+        return process.ExitCode;
+    }
+}
